继承后对象的创建

  • 子类的对象是由子类、父类共同完成的。

 | 技术 
package string;

public class StringDemo {
    public static void main(String[] args) {
        String str = "hello";
        for (int i = 0; i < str.length(); i++) {
            /*
                h e l l o
            i = 0 1 2 3 4
             */
            char ch = str.charAt(i);// charAt:提取字符串中某个索引位置处的字符
            System.out.println(ch);
            /*
            * 控制台输出：
            * h
            * e
            * l
            * l
            * o
            */
        }
    }
